<script>
  const audio  = document.getElementById('audio');
  const play   = document.getElementById('play');
  const pause  = document.getElementById('pause');
  const replay = document.getElementById('replay');
  const back5  = document.getElementById('back5');
  const fwd5   = document.getElementById('fwd5');
  const seek   = document.getElementById('seek');
  const volume = document.getElementById('volume');
  const rate   = document.getElementById('rate');
  const timeEl = document.getElementById('time');

  // ユーティリティ：mm:ss
  function fmt(t) {
    if (!isFinite(t) || t < 0) return "--:--";
    const m = Math.floor(t / 60);
    const s = Math.floor(t % 60);
    return `${String(m).padStart(2,'0')}:${String(s).padStart(2,'0')}`;
  }

  // 初期化：メタデータロード後に合計時間を表示
  audio.addEventListener('loadedmetadata', () => {
    timeEl.textContent = `${fmt(audio.currentTime)} / ${fmt(audio.duration)}`;
  });

  // 再生・一時停止
  play.addEventListener('click', () => audio.play());
  pause.addEventListener('click', () => audio.pause());

  // Replay（頭出し→即再生）
  replay.addEventListener('click', () => {
    audio.currentTime = 0;
    audio.play();
  });

  // 5秒戻す／進める（境界チェック）
  back5.addEventListener('click', () => {
    audio.currentTime = Math.max(0, audio.currentTime - 5);
  });
  fwd5.addEventListener('click', () => {
    const end = isFinite(audio.duration) ? audio.duration : audio.currentTime + 5;
    audio.currentTime = Math.min(end, audio.currentTime + 5);
  });

  // 再生位置の表示＆シークバー反映
  audio.addEventListener('timeupdate', () => {
    timeEl.textContent = `${fmt(audio.currentTime)} / ${fmt(audio.duration)}`;
    if (isFinite(audio.duration) && audio.duration > 0) {
      seek.value = Math.round((audio.currentTime / audio.duration) * 1000);
    }
  });

  // シークバーからの操作（ドラッグ中の更新も許容）
  seek.addEventListener('input', () => {
    if (isFinite(audio.duration) && audio.duration > 0) {
      const t = (seek.value / 1000) * audio.duration;
      audio.currentTime = t;
    }
  });

  // 音量・速度
  volume.addEventListener('input', () => { audio.volume = parseFloat(volume.value); });
  rate.addEventListener('change', () => { audio.playbackRate = parseFloat(rate.value); });

  // 再生終了時のUI（終了後に Replay を案内したい場合はボタン状態を更新）
  audio.addEventListener('ended', () => {
    // 任意：終了後は再生ボタンの文言を変えるなど
    // play.textContent = "▶︎ 再生";
  });

  // 右クリック抑止（要素全体でも設定済みだが念のため）
  document.querySelector('.player').addEventListener('contextmenu', (e) => e.preventDefault());

  // キーボード操作（アクセシビリティ向上）
  document.addEventListener('keydown', (e) => {
    // Space / K: 再生・一時停止
    if (e.code === 'Space' || e.key.toLowerCase() === 'k') {
      e.preventDefault();
      audio.paused ? audio.play() : audio.pause();
    }
    // J/L: 5秒移動
    if (e.key.toLowerCase() === 'j') back5.click();
    if (e.key.toLowerCase() === 'l') fwd5.click();
    // R: Replay
    if (e.key.toLowerCase() === 'r') replay.click();
  });
</script>
